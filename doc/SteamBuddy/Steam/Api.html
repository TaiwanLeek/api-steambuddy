def game_name(appid)
  url = 'https://store.steampowered.com/api/appdetails'
  parameter = {
    query: {
      appids: appid
    }
  }
  response = HTTParty.get(url, parameter)
  body = JSON.parse(response.body)
  game_body = body[appid.to_s] if body
  game_body['data']['name'] if game_body && game_body['success']
end
